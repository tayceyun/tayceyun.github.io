  if (n === 0) return 1;

  while (n > 0) {
    let current = n % 2;
    arr.push(current);
    n = Math.floor(n / 2);

  while (arr.length) {
    str.unshift(arr[arr.length - 1] === 0 ? 1 : 0);
    arr.pop();
